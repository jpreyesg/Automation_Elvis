p_1 = (p_1) ? p_1 : "null";
var p_2 = data.checkin;
p_2 = (p_2) ? p_2.replace(/\/+/g, '-') : "null";
var p_3 = data.checkout;
p_3 = (p_3) ? p_3.replace(/\/+/g, '-') : "null";
var p_4 = data.adults;
p_4 = (p_4) ? p_4 : 0;
var p_5 = data.children;
p_5 = (p_5) ? p_5 : 0;
var url = "";
if (p_1 != "null") {
url += "/" + p_1
}
return url + "/" + p_2 + "/" + p_3 + "/" + p_4 + "/" + p_5
}
